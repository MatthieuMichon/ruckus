267 def buildRogueFile(zipName, cfg, ver, relName, relData, imgList):
268  print("\nFinding Rogue Files...")
269  pList = selectFiles(cfg, relData, 'RoguePackages')
270  cList = selectFiles(cfg, relData, 'RogueConfig')
271 
272  if len(pList) == 0:
273  raise Exception(f"Invalid release config. Rogue packages list is empty!")
274 
275  if not 'TopPackage' in cfg or cfg['TopPackage'] is None:
276  raise Exception("Invalid release config. TopPackage is not defined!")
277 
278  # setuptools version creates and installs a .egg file which will not work with
279  # our image and config data! Use distutils version for now.
280  setupPy = "\n\nfrom distutils.core import setup\n\n"
281  #setupPy = "\n\nfrom setuptools import setup\n\n"
282  setupPy += "setup (\n"
283  setupPy += f" name='{relName}',\n"
284  setupPy += f" version='{ver}',\n"
285  setupPy += " packages=[\n"
286 
287  topInit = cfg['TopPackage'] + '/__init__.py'
288  topPath = None
289 
290  with zipfile.ZipFile(zipName,'w') as zf:
291  print(f"\nCreating Rogue zipfile {zipName}")
292 
293  for e in pList:
294  dst = e['subPath']
295 
296  # Don't add raw version of TopPackage/__init__.py
297  if e['subPath'] == topInit:
298  topPath = e['fullPath']
299  else:
300  zf.write(e['fullPath'],dst)
301 
302  if e['type'] == 'folder':
303  setupPy += " '{}',\n".format(dst)
304 
305  setupPy += " ],\n"
306 
307  for e in cList:
308  dst = cfg['TopPackage'] + '/config/' + e['subPath']
309  zf.write(e['fullPath'],dst)
310 
311  for e in imgList:
312  dst = cfg['TopPackage'] + '/images/' + os.path.basename(e)
313  zf.write(e,dst)
314 
315  # Generate setup.py payload
316  setupPy += " package_data={'" + cfg['TopPackage'] + "':['config/*','images/*']}\n"
317  setupPy += ")\n"
318 
319  with zf.open('setup.py','w') as sf: sf.write(setupPy.encode('utf-8'))
320 
321  with open(topPath,'r') as f:
322  newInit = ""
323 
324  for line in f:
325  if (not 'import os' in line) and \
326  (not '__version__' in line) and \
327  (not 'ConfigDir' in line) and \
328  (not 'ImageDir' in line): newInit += line
329 
330  # Append new lines
331  newInit += "\n\n"
332  newInit += "##################### Added by release script ###################\n"
333  newInit += "import os\n"
334  newInit += f"__version__ = '{ver}'\n"
335  newInit += "ConfigDir = os.path.dirname(__file__) + '/config'\n"
336  newInit += "ImageDir = os.path.dirname(__file__) + '/images'\n"
337  newInit += "#################################################################\n"
338 
339  with zf.open(topInit,'w') as f:
340  f.write(newInit.encode('utf-8'))
