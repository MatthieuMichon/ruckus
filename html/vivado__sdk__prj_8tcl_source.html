31  # Configure the release build
32  sdk get_build_config -app app_0 build-config release

34  foreach sdkLib ${SDK_LIB} {

37  sdk get_build_config -app app_0 -add include-path ${sdkLib}
38  exec ln -s ${sdkLib} ${softLink}

40 } else {
